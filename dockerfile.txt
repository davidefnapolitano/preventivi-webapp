# ----------------------------
# Stage 1: immagine base con Python
# ----------------------------
FROM python:3.10-slim

# Evita prompt interattivi
ENV DEBIAN_FRONTEND=noninteractive

# Aggiorna e installa LibreOffice headless + dipendenze minime
RUN apt-get update && \
    apt-get install -y --no-install-recommends \
      libreoffice-core \
      libreoffice-writer \
      libreoffice-calc \
      libreoffice-common \
      xz-utils \
      unzip \
      fontconfig \
      libglu1-mesa \
      libxrender1 \
      libxtst6 \
      libxi6 \
      libxrandr2 \
      libxfixes3 \
      libxdamage1 \
      libpango1.0-0 \
      libglib2.0-0 \
      && rm -rf /var/lib/apt/lists/*

# Imposta la working directory
WORKDIR /app

# Copia i file di requirements e installa le dipendenze Python
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Copia il codice dell’applicazione
COPY . .

# Crea la cartella out/ (per i file temporanei)
RUN mkdir -p /app/out

# Espone la porta che Flask utilizzerà (Gunicorn)
EXPOSE 5000

# Comando di avvio: usa gunicorn per esporre l’app Flask
CMD ["gunicorn", "--bind", "0.0.0.0:5000", "app:app"]
